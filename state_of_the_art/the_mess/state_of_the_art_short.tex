% VISION TRANSFORMERS

\begin{frame}{State of the Art: Vision Transformers for Medical Imaging}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Evolution}
                \begin{itemize}
                    \item \textbf{ViT (2020):} First pure transformer for images
                    \item \textbf{Swin (2021):} Shifted windows, $\mathcal{O}(n)$ complexity
                    \item \textbf{UNETR (2022):} ViT encoder + CNN decoder
                    \item \textbf{Swin UNETR (2022):} Current SOTA for 3D medical
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Swin UNETR on BraTS 2021}
                \begin{center}
                    \begin{tabular}{lc}
                        \toprule
                        \textbf{Region} & \textbf{Dice} \\
                        \midrule
                        Whole Tumor & 0.929 \\
                        Tumor Core & 0.891 \\
                        Enhancing & 0.854 \\
                        \bottomrule
                    \end{tabular}
                \end{center}
            \end{block}
            
            \begin{alertblock}{Our Choice}
                \textbf{Swin UNETR} for segmentation + feature extraction
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

% DIFFUSION MODELS
% =============================================================================
\begin{frame}{State of the Art: Diffusion Models for Video Generation}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Evolution}
                \begin{itemize}
                    \item \textbf{DDPM (2020):} Denoising diffusion foundation
                    \item \textbf{LDM (2022):} Latent space diffusion (64$\times$ faster)
                    \item \textbf{Video LDM (2023):} Temporal layers for video
                    \item \textbf{Stable Video Diffusion:} Production-ready
                \end{itemize}
            \end{block}
            
            \begin{block}{Core Formula}
                \begin{center}
                    $p_\theta(\mathbf{x}_{t-1} | \mathbf{x}_t) = \mathcal{N}(\boldsymbol{\mu}_\theta, \sigma_t^2\mathbf{I})$
                \end{center}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Video LDM Architecture}
                \begin{itemize}
                    \item Spatial layers (per frame)
                    \item Temporal attention (across frames)
                    \item Conditioning via cross-attention
                \end{itemize}
            \end{block}
            
            \begin{alertblock}{Our Goal}
                Generate counterfactual disease progression: \\
                \textit{``What if treatment A vs B?''}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}
